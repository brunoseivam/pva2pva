77 void add_column(void *pvoid, const char* fname, const char* label, pvd::ScalarType type)
78 {
79  multiArrayCommonRecord *prec = (multiArrayCommonRecord*)pvoid;
80 
81  if(prec->rpvt->finished)
82  throw std::logic_error("can't add_column() after init_record()");
83 
84  if(prec->rpvt->entries_map.find(fname)!=prec->rpvt->entries_map.end())
85  throw std::logic_error("Can't add_column() with duplicate name");
86 
87  prec->rpvt->entries.push_back(EntryStorage());
88  EntryStorage& ent = prec->rpvt->entries.back();
89  prec->rpvt->entries_map[fname] = &ent;
90 
91  ent.name = fname;
92  ent.label = label ? label : fname;
93  ent.type = type;
94 }

114

122 
123  if(prec->lay==menuMultiArrayLayoutTable) {
124  pvd::PVScalarArrayPtr arr(prec->val->getSubFieldT<pvd::PVScalarArray>(ent.name));
125  arr->putFrom(cdata);
126  prec->vld.set(arr->getFieldOffset());
127 
128  } else if(prec->lay==menuMultiArrayLayoutComposite) {
129  pvd::PVStructurePtr base(prec->val->getSubFieldT<pvd::PVStructure>(ent.name));
130 
131  pvd::PVScalarArrayPtr arr(base->getSubFieldT<pvd::PVScalarArray>("value"));
132  arr->putFrom(cdata);
133  prec->vld.set(arr->getFieldOffset());
134 
135  if(pmeta) {
136  storeAlarm(*base, prec->vld, pmeta->sevr, pmeta->amsg ? pmeta->amsg : "");
137  storeTime(*base, prec->vld, pmeta->time, pmeta->utag);
138  }

190 long init_record(struct dbCommon *pcommon, int pass)
191 {
192  multiArrayCommonRecord *prec = (multiArrayCommonRecord*)pcommon;
193  TRY {
194  if(pass==0) {
195  prec->rpvt = new RPvt;
196  new (&prec->val) pvd::PVStructurePtr();
197  new (&prec->vld) pvd::BitSet;
198 
199  // caller may now add_column()
200 
201  return 0;
202 
203  } else if(pass==1) {
204  // caller has made all calls to add_column()
205 
206  prec->rpvt->finished = true;
207 
208  const pvd::StandardFieldPtr& sfld(pvd::getStandardField());
209 
210  pvd::FieldBuilderPtr builder = pvd::FieldBuilder::begin();
211 
212  if(prec->lay==menuMultiArrayLayoutTable) {
213 
214  pvd::shared_vector<std::string> labels;
215  labels.reserve(prec->rpvt->entries.size());
216 
217  builder = builder->setId("epics:nt/NTTable:1.0")
218  ->addArray("labels", pvd::pvString)
219  ->addNestedStructure("value");
220 
221  for(RPvt::entries_t::const_iterator it(prec->rpvt->entries.begin()),
222  end(prec->rpvt->entries.end()); it!=end; ++it)
223  {
224  const EntryStorage& ent = *it;
225 
226  builder = builder->addArray(ent.name, ent.type);
227  labels.push_back(ent.label);
228  }
229 
230  builder = builder->endNested()
231  ->add("alarm", sfld->alarm())
232  ->add("timeStamp", sfld->timeStamp());
233 
234  prec->val = builder->createStructure()->build();
235 
236  pvd::PVStringArrayPtr flabels(prec->val->getSubFieldT<pvd::PVStringArray>("labels"));
237  flabels->replace(pvd::freeze(labels));
238  prec->vld.set(flabels->getFieldOffset());
239 
240  pvd::PVStructurePtr value(prec->val->getSubFieldT<pvd::PVStructure>("value"));
241 
242  for(RPvt::entries_t::const_iterator it(prec->rpvt->entries.begin()),
243  end(prec->rpvt->entries.end()); it!=end; ++it)
244  {
245  const EntryStorage& ent = *it;
246 
247  if(!ent.value.empty())
248  continue;
249 
250  pvd::PVScalarArrayPtr fld(value->getSubFieldT<pvd::PVScalarArray>(ent.name));
251  fld->putFrom(ent.value);
252  prec->vld.set(fld->getFieldOffset());
253  }
254 
255  } else if(prec->lay==menuMultiArrayLayoutComposite) {
256 
257  for(RPvt::entries_t::const_iterator it(prec->rpvt->entries.begin()),
258  end(prec->rpvt->entries.end()); it!=end; ++it)
259  {
260  const EntryStorage& ent = *it;
261 
262  pvd::StructureConstPtr substruct(sfld->scalarArray(ent.type, "alarm,timeStamp"));
263 
264  builder = builder->add(ent.name, substruct);
265  }
266 
267  prec->val = builder->createStructure()->build();
268 
269  for(RPvt::entries_t::iterator it(prec->rpvt->entries.begin()),
270  end(prec->rpvt->entries.end()); it!=end; ++it)
271  {
272  EntryStorage& ent = *it;
273 
274  if(!ent.value.empty())
275  continue;
276 
277  pvd::PVStructurePtr base(prec->val->getSubFieldT<pvd::PVStructure>(ent.name));
278 
279  pvd::PVScalarArrayPtr fld(base->getSubFieldT<pvd::PVScalarArray>("value"));
280  fld->putFrom(ent.value);
281  ent.value = pvd::shared_vector<const void>(); // clear
282  prec->vld.set(fld->getFieldOffset());
283 
284  // TODO initial alarm/time
285  }
286 
287  } else {
288  errlogPrintf("%s.LAY: out of range value %d\n", prec->name, prec->lay);
289  (void)recGblSetSevrMsg(prec, COMM_ALARM, INVALID_ALARM, "Bad LAY %d", prec->lay);
290  return -1;
291  }
292 
293  return 0;
294  }
295  }CATCH()
296  return -1;
297 }

302 
303  if(monitor_mask&DBE_ALARM) {

306  storeAlarm(*prec->val, prec->vld, prec->sevr, prec->amsg);
307 
308  } else if(prec->lay==menuMultiArrayLayoutComposite) {
309  // no top level alarm

311  }
312 
313  if(prec->lay==menuMultiArrayLayoutTable) {
314  storeTime(*prec->val, prec->vld, prec->time, prec->utag);
315 
316  } else if(prec->lay==menuMultiArrayLayoutComposite) {
317  // no top level timestamp
318  }
319

321  db_post_events(prec, &prec->val, monitor_mask);
322  }

361  TRY {
362 
363  if(!prec->val)

366  if(p->vtype==&vfPVStructure) {

373  return 0;

375

377  VSharedStructure *pstr = (VSharedStructure*)p;
378  if(dbGetFieldIndex(paddr)==multiArrayCommonRecordVAL) {
379  *pstr->value = prec->val->getStructure();
380  return 0;
381  }
382  }

384  }CATCH()
